## 字符串

```s
# 时间复杂度是 O(N*N)，每次拿到新的字符都要和之前的做比较
class Solution {
    func lengthOfLongestSubstring(_ s: String) -> Int {
        var tmp = ""
        var count = 0
        for ch in s {
            if let index = tmp.firstIndex(of: ch) {
                tmp.removeSubrange(tmp.startIndex...index)
            }
            tmp.append(ch)
            count = max(count, tmp.count)
        }
        return count
    }
}
```
