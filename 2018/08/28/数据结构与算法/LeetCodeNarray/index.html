加一

数组中每个元素都是单个正整数
特殊情况是在当前数为 9 ，需要进位时改变高位数字

class Solution {
    func plusOne(_ digits: [Int]) -> [Int] {
        var nums1 = digits
        let count = nums1.count
        var curIndex = count - 1
        let curEle = nums1[curIndex]

        if curEle != 9{
            nums1[curIndex] += 1
        }else{
            nums1[curIndex] = 0
            while curIndex > 0 {
                curIndex -= 1
                if nums1[curIndex] != 9{
                    nums1[curIndex] += 1
                    break
                }else{
                    nums1[curIndex] = 0
                }
            }

            if nums1[0] == 0 {
                nums1.insert(1, at: 0)
            }
        }
        return nums1`
